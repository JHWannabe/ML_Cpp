--------------------------------------------
Command Line Arguments:
--------------------------------------------
Options:
  --help                                   produce help message
  --dataset arg                            dataset name
  --class_list arg (=list/VOC2012.txt)     file name in which class names are listed
  --anchor_list arg (=cfg/anchor.txt)      file name in which anchors are listed
  --resize_list arg (=cfg/resize.txt)      file name in which resizes are listed
  --class_num arg (=20)                    total classes
  --size arg (=608)                        image width and height
  --prob_thresh arg (=0.100000001)         threshold of simultaneous probability with confidence and class score
  --ignore_thresh arg (=0.699999988)       threshold of IoU between no object anchor and ground truth for excluding loss of 'no object confidence term'
  --nms_thresh arg (=0.5)                  threshold of IoU between bounding boxes in Non-Maximum Suppression
  --nc arg (=3)                            input image channel : RGB=3, grayscale=1
  --na arg (=3)                            the number of anchor in each grid
  --scales arg (=3)                        the number of predicted map scales
  --gpu_id arg (=0)                        cuda device : 'x=-1' is cpu device
  --seed_random arg (=0)                   whether to make the seed of random number in a random
  --seed arg (=0)                          seed of random number
  --train arg (=0)                         training mode on/off
  --train_in_dir arg (=trainI)             training input image directory : ./datasets/<dataset>/<train_in_dir>/<image files>
  --train_out_dir arg (=trainO)            training output image directory : ./datasets/<dataset>/<train_out_dir>/<annotation files>
  --epochs arg (=200)                      training total epoch
  --batch_size arg (=32)                   training batch size
  --train_load_epoch arg                   epoch of model to resume learning
  --save_epoch arg (=20)                   frequency of epoch to save model and optimizer
  --augmentation arg (=1)                  data augmentation mode on/off
  --jitter arg (=0.29999999999999999)      the distortion of image shifting
  --flip_rate arg (=0.5)                   frequency to flip
  --scale_rate arg (=0.5)                  frequency to scale
  --blur_rate arg (=0.5)                   frequency to blur
  --brightness_rate arg (=0.5)             frequency to change brightness
  --hue_rate arg (=0.5)                    frequency to change hue
  --saturation_rate arg (=0.5)             frequency to change saturation
  --shift_rate arg (=0.5)                  frequency to shift
  --crop_rate arg (=0.5)                   frequency to crop
  --valid arg (=0)                         validation mode on/off
  --valid_in_dir arg (=validI)             validation input image directory : ./datasets/<dataset>/<valid_in_dir>/<image files>
  --valid_out_dir arg (=validO)            validation output image directory : ./datasets/<dataset>/<valid_out_dir>/<annotation files>
  --valid_batch_size arg (=1)              validation batch size
  --valid_freq arg (=1)                    validation frequency to training epoch
  --test arg (=0)                          test mode on/off
  --test_in_dir arg (=testI)               test input image directory : ./datasets/<dataset>/<test_in_dir>/<image files>
  --test_out_dir arg (=testO)              test output image directory : ./datasets/<dataset>/<test_out_dir>/<annotation files>
  --test_load_epoch arg (=latest)          training epoch used for testing
  --test_result_dir arg (=test_result)     test result directory : ./<test_result_dir>
  --detect arg (=0)                        detection mode on/off
  --detect_dir arg (=detect)               detection image directory : ./datasets/<dataset>/<detect_dir>/<image files>
  --detect_load_epoch arg (=latest)        training epoch used for detection
  --detect_result_dir arg (=detect_result) detection result directory : ./<detect_result_dir>
  --demo arg (=0)                          demo mode on/off
  --cam_num arg (=0)                       the port number of camera
  --movie arg                              movie file name : ./<movie>
  --window_w arg (=1920)                   width of window size
  --window_h arg (=1080)                   height of window size
  --demo_load_epoch arg (=latest)          training epoch used for demo
  --lr_init arg (=0.00100000005)           learning rate in the initial iteration
  --lr_base arg (=0.00999999978)           learning rate in the base iteration
  --lr_decay1 arg (=0.00100000005)         learning rate in the decay 1 iteration
  --lr_decay2 arg (=9.99999975e-05)        learning rate in the decay 2 iteration
  --momentum arg (=0.899999976)            momentum in SGD of optimizer method
  --weight_decay arg (=0.000500000024)     weight decay in SGD of optimizer method
  --Lambda_coord arg (=1)                  the multiple of coordinate term
  --Lambda_object arg (=1)                 the multiple of object confidence term
  --Lambda_noobject arg (=1)               the multiple of no object confidence term
  --Lambda_class arg (=1)                  the multiple of class term
